import moment from 'moment';

/**
 * Formats given date string
 * Will convert date to local timezone first
 *
 * @param date {string}
 * @param utc {boolean}
 * @param format {string}
 * @returns {string}
 */
const format = ({ date , utc = false, format = 'D MMM YYYY' } = {}) => {
	return moment(date).local().format(format)
}

export default {
	format,
};
